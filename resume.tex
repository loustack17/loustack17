% ---------------------------------------------------------
% Fix Awesome-CV: remove extra blank space when #5 is empty
% Source: TeX StackExchange / Awesome-CV community
% ---------------------------------------------------------
\renewcommand*{\cventry}[5]{%
  \vspace{-1.0mm}%
  \setlength\tabcolsep{0pt}%
  \setlength{\extrarowheight}{0pt}%
  \begin{tabular*}{\textwidth}{@{\extracolsep{\fill}} L{\textwidth - 4.5cm} R{4.5cm}}
    \ifempty{#2#3}
      {\entrypositionstyle{#1} & \entrydatestyle{#4} \\}
      {\entrytitlestyle{#2} & \entrylocationstyle{#3} \\[-1.5mm]
       \entrypositionstyle{#1} & \entrydatestyle{#4} \\}
    % ---- only render description row when #5 is NOT empty ----
    \if\relax\detokenize{#5}\relax\else
      \multicolumn{2}{L{\textwidth}}{\descriptionstyle{#5}}%
    \fi
  \end{tabular*}%
}
